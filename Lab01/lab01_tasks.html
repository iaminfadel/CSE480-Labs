# Load character image
img = cv2.imread('./image.png')
if img is not None:
    # Resize to 200x200
    photo = cv2.resize(img, (200, 200))

    # Create ID card canvas (white background)
    id_card = np.ones((250, 400, 3), dtype=np.uint8) * 255
    
    # Place photo
    id_card[25:225, 25:225] = photo

    # Add text
    font = cv2.FONT_HERSHEY_SIMPLEX
    cv2.putText(id_card, "Name: Amin", (240, 60), font, 0.7, (0, 0, 0), 2)
    cv2.putText(id_card, "Series: ASU", (240, 100), font, 0.6, (50, 50, 50), 1)
    cv2.putText(id_card, "ID: 67-12345", (240, 140), font, 0.6, (100, 0, 0), 1)

    # Add a border
    id_card = cv2.copyMakeBorder(id_card, 5, 5, 5, 5, cv2.BORDER_CONSTANT, value=[0, 0, 0])

    # Save and display
    cv2.imwrite('./character_id_card.jpg', id_card)
    print("Task 1 Complete: Saved character_id_card.jpg")
    show_image('./character_id_card.jpg')
else:
    print("Error: ./image.png not found.")

Task 2: Social Media Frame¶

Load character image, flip it horizontally, and draw a colored rectangle frame with simulated rounded edges.

In [7]:
img = cv2.imread('./image.png')
if img is not None:
    # Flip horizontal
    flipped = cv2.flip(img, 1)

    # Draw frame
    h, w, _ = flipped.shape
    color = (255, 100, 100)
    thickness = 20
    cv2.rectangle(flipped, (thickness//2, thickness//2), (w - thickness//2, h - thickness//2), color, thickness)
    
    # Rounded corners simulation
    radius = 30
    cv2.circle(flipped, (radius, radius), radius, color, -1)
    cv2.circle(flipped, (w - radius, radius), radius, color, -1)
    cv2.circle(flipped, (radius, h - radius), radius, color, -1)
    cv2.circle(flipped, (w - radius, h - radius), radius, color, -1)

    # Save and display
    cv2.imwrite('./profile_frame.jpg', flipped)
    print("Task 2 Complete: Saved profile_frame.jpg")
    show_image('./profile_frame.jpg')
else:
    print("Error: image.png not found.")

Task 3: Dramatic Transition¶

Blend two images gradually using cv2.addWeighted().

img1 = cv2.imread('./image.png')
img2 = cv2.imread('./image2.png')

if img1 is not None and img2 is not None:
    # Resize to match
    img1 = cv2.resize(img1, (500, 500))
    img2 = cv2.resize(img2, (500, 500))

    # Blend (showing the middle frame of the 5 transition frames)
    alpha = 0.5
    blend = cv2.addWeighted(img1, 1 - alpha, img2, alpha, 0)

    # Save and display
    cv2.imwrite('./transition_scene.jpg', blend)
    print("Task 3 Complete: Saved transition_scene.jpg")
    show_image('./transition_scene.jpg')
else:
    print("Error: Images not found.")
